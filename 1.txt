
console.log(" " + 1 + 0) // 10 --> Сначала преобразование в строку, затем сложение символов
console.log(" " - 1 + 0) // -1 --> Присваивается минус строке, затем единица складывается с нулём
console.log(true + false) // 1 --> Преобразование в числа, затем выполняется сложение
console.log(6 / "3") // 2 --> Преобразование в числа, затем выполняется деление
console.log("2" * "3") // 6 --> Преобразование в числа, затем выполняется умножение
console.log(4 + 5 + "px") // 9px --> Сложение чисел, затем добавление текста
console.log("$" + 4 + 5) // $45 --> Текстовый формат вывода, вывод символов
console.log("4" - 2) // 2 --> Преобразование в числа, затем выполняется вычитание
console.log("4px" - 2) // NaN --> Невозможно преобразовать 4px в число, поэтому операция невозможна
console.log(" -9 " + 5) // -9 5 --> Сложение строки с символом
console.log(" -9 " - 5) // -14 --> Преобразование в числа, затем выполняется вычитание
console.log(null + 1) // 1 --> null преобразуется в 0, затем выполняется сложение
console.log(undefined + 1) // NaN --> undefined невозможно сложить с числом
console.log(" \t \n" - 2) // -2 --> Символы обрезаются, из непустой строки считывается число
console.log(5 > 4) // true --> Сравнение чисел по значению
console.log("ананас" > "яблоко") // false --> Происходит сравнение по Unicode
console.log("2" > "12") // true --> Первый символ у левого операнда больше первого у правого
console.log(undefined == null) // true --> Значения операндов равны
console.log(undefined === null) // false --> Значения операндов равны, но разные типы
console.log(null == "\n0\n") // false --> null может быть равен только undefined
console.log(null === +"\n0\n") // false --> Сравниваются разные типы
